# Get all workload objects in the namespace
kubectl get all -n k8s-program
# Why does `kubectl get all -n k8s-program` show Pods and ReplicaSets?
# A Deployment doesn’t create Pods directly; it creates a ReplicaSet, and the ReplicaSet creates Pods.
# Similarly, a StatefulSet manages its own Pods and dynamically creates PVCs from its volumeClaimTemplates.

kubectl get pods -n k8s-program
kubectl get svc -n k8s-program
kubectl get deploy -n k8s-program
kubectl get rs -n k8s-program
kubectl get sts -n k8s-program

# Sub-task 3: Persistent volumes

# Inspect PersistentVolumeClaim and PersistentVolume (both should show STATUS=Bound and PersistentVolumeClaim should show the correct PersistentVolume in the "VOLUME" column)
kubectl get pvc songs-pvc -o wide -n k8s-program
kubectl get pv songs-pv

# Write a file to the mounted volume inside the container, then list the directory
kubectl exec deploy/songs-ms-dpl -c songs-service -n k8s-program -- sh -lc 'echo hello > /data/test.txt && ls -l /data'

# Force a redeploy (scale to zero and back to one) and wait for readiness
kubectl scale deploy/songs-ms-dpl --replicas=0 -n k8s-program
kubectl get pods -n k8s-program
kubectl scale deploy/songs-ms-dpl --replicas=1 -n k8s-program
kubectl get pods -n k8s-program

# Read the file back (new Pod, same PersistentVolumeClaim)
kubectl exec deploy/songs-ms-dpl -c songs-service -n k8s-program -- sh -lc 'cat /data/test.txt'
