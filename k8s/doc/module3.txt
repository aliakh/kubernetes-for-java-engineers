# Verify that Helm is installed
helm version

# Scaffold a new Helm chart named "helm"
helm create helm

# Render the chart locally into Kubernetes YAML (no resources are created in the cluster) and redirect output to a file for review
helm template release1 ./helm > manifests.yaml

# Install the chart as a Helm release named "release1" into the Kubernetes cluster in namespace "k8s-program"
helm install release1 ./helm --namespace k8s-program --create-namespace

# Check that by default each microservice has 2 Pods
kubectl get pods -n k8s-program

# Check that the application is working
curl.exe -sL -X POST "http://localhost:30080/resources/file" -H "Content-Type: multipart/form-data" -F "file=@mp3/resource.mp3"
curl.exe -sL "http://localhost:30081/songs/1"

# Upgrade the chart into the Kubernetes cluster to set replica-count to non-default values
# Important:
# If we try to set namespace to a new value (e.g. "k8s-program2"), then the existing release will not be moved from k8s-program to k8s-program2.
# Instead, it upgrades release1 in k8s-program2 if it exists; otherwise, it installs a new release named release1 in k8s-program2.
# The original release in k8s-program remains unchanged unless you uninstall it explicitly.
helm upgrade --install release1 ./helm --namespace k8s-program --set resources_ms_dpl.replicas=3 --set songs_ms_dpl.replicas=3

# Check that after upgrade each microservice has 3 Pods
kubectl get pods -n k8s-program

# Check that the application is still working
curl.exe -sL -X POST "http://localhost:30080/resources/file" -H "Content-Type: multipart/form-data" -F "file=@mp3/resource.mp3"
curl.exe -sL "http://localhost:30081/songs/2"
