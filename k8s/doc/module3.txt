# Verify that Helm is installed
helm version

# Scaffold a new Helm chart named "helm"
helm create helm

# Render the chart locally into Kubernetes YAML (no resources are created in the cluster) and redirect output to a file for review
helm template release1 ./helm > manifests.yaml

# Install the chart as a Helm release named "release1" into the Kubernetes cluster in namespace "k8s-program"
helm install release1 ./helm --namespace k8s-program --create-namespace

# Check that by default each microservice has 2 Pods
kubectl get pods -n k8s-program

# Check that the application is working
curl.exe -sL -X POST "http://localhost:30080/resources/file" -H "Content-Type: multipart/form-data" -F "file=@mp3/resource.mp3"
curl.exe -sL "http://localhost:30081/songs/1"

# Upgrade the release in the Kubernetes cluster to set the replica count to non-default values.
# Important:
# - Helm releases are namespaced. The flag --namespace k8s-program selects the namespace where Helm looks for the release record.
# - Changing --namespace does not move an existing release from k8s-program to another namespace (e.g., k8s-program2).
# - If you run helm upgrade with --namespace k8s-program2, Helm will attempt to upgrade release1 in k8s-program2:
#   - it will succeed only if release1 already exists in k8s-program2;
#   - otherwise, it will fail (because this command does not include --install).
# - The original release in k8s-program remains unchanged unless you upgrade or uninstall it explicitly.
helm upgrade release1 ./helm --namespace k8s-program --set resources_ms_dpl.replicas=3 --set songs_ms_dpl.replicas=3

# Check that after upgrade each microservice has 3 Pods
kubectl get pods -n k8s-program

# Check that the application is still working
curl.exe -sL -X POST "http://localhost:30080/resources/file" -H "Content-Type: multipart/form-data" -F "file=@mp3/resource.mp3"
curl.exe -sL "http://localhost:30081/songs/2"

# Check that a ConfigMap has labels: currentDate rendered by the helper template in _helpers.tpl at Helm render time, and version from Chart.yaml
kubectl describe configmap resources-ms-config-cm -n k8s-program
